-- Vari√°veis e fun√ß√µes
local camera = workspace.CurrentCamera
local originalCameraSubject = camera.CameraSubject
local loopGoto = false
local noclipEnabled = false
local levitating = false
local spinning = false
local xrayEnabled = false

-- Fun√ß√£o para procurar jogador pelo nome
local function findPlayerByPartialName(partialName)
    partialName = partialName:lower()
    for _, plr in ipairs(game.Players:GetPlayers()) do
        if plr.Name:lower():sub(1, #partialName) == partialName then
            return plr
        end
    end
    return nil
end

local function executeCommand(command)
    local args = command:split(" ")

    if args[1] == ":cmds" then
        cmdsGui.Enabled = true
    elseif args[1] == ":tp" and args[2] then
        local target = findPlayerByPartialName(args[2])
        if target then
            player.Character:MoveTo(target.Character.HumanoidRootPart.Position)
        end
    elseif args[1] == ":view" and args[2] then
        local target = findPlayerByPartialName(args[2])
        if target and target.Character then
            camera.CameraSubject = target.Character:FindFirstChild("Humanoid")
        end
    elseif args[1] == ":unview" then
        camera.CameraSubject = originalCameraSubject
    elseif args[1] == ":loopgoto" and args[2] then
        local target = findPlayerByPartialName(args[2])
        if target then
            loopGoto = true
            while loopGoto and task.wait(0.1) do
                if player.Character and target.Character then
                    player.Character:MoveTo(target.Character.HumanoidRootPart.Position)
                end
            end
        end
    elseif args[1] == ":unloopgoto" then
        loopGoto = false
    elseif args[1] == ":noclip" then
        noclipEnabled = true
    elseif args[1] == ":clip" then
        noclipEnabled = false
    elseif args[1] == ":spin" and args[2] then
        local spinSpeed = tonumber(args[2])
        if spinSpeed then
            spinning = true
            while spinning do
                if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                    local humanoidRootPart = player.Character.HumanoidRootPart
                    humanoidRootPart.CFrame = humanoidRootPart.CFrame * CFrame.Angles(0, math.rad(spinSpeed), 0)
                end
                task.wait(0.05)
            end
        end
    elseif args[1] == ":unspin" then
        spinning = false
    elseif args[1] == ":levitacao" then
        levitating = true
        while levitating do
            if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
                local humanoidRootPart = player.Character.HumanoidRootPart
                humanoidRootPart.Anchored = true
                humanoidRootPart.Position = humanoidRootPart.Position + Vector3.new(0, 0.5, 0) -- Eleva um pouco
            end
            task.wait(0.1)
        end
    elseif args[1] == ":unlevitacao" then
        levitating = false
        if player.Character and player.Character:FindFirstChild("HumanoidRootPart") then
            local humanoidRootPart = player.Character.HumanoidRootPart
            humanoidRootPart.Anchored = false
        end
    elseif args[1] == ":xray" then
        xrayEnabled = true
        for _, part in ipairs(workspace:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Material = Enum.Material.SmoothPlastic
                part.Transparency = 0.5
            end
        end
    elseif args[1] == ":unxray" then
        xrayEnabled = false
        for _, part in ipairs(workspace:GetDescendants()) do
            if part:IsA("BasePart") then
                part.Material = Enum.Material.SmoothPlastic
                part.Transparency = 0
            end
        end
    end
end

button.MouseButton1Click:Connect(function()
    executeCommand(textbox.Text)
end)

-- Comandos via chat
game.Players.LocalPlayer.Chatted:Connect(function(message)
    if message:sub(1, 1) == ":" then
        executeCommand(message)
    end
end)
